public class Item {
    private String name;
    private String color;
    private int weight;
    private int thrownAttackValue;
    
    public Item(){
        name = "mystery";
        color = "black";
        weight = 0;
        thrownAttackValue = 0;
    }
    public Item(String name, String color, int weight, int thrownAttackValue){
        this.name = name;
        this.color = color;
        this.weight = weight;
        this.thrownAttackValue = thrownAttackValue;
    }
    
    public String getName() {
        return name;
    }
    public String getColor() {
        return color;
    }
    public int getWeight() {
        return weight;
    }
    public int getThrownAttackValue() {
        return thrownAttackValue;
    }
    public void setName(String name){
        this.name = name;
    }
    public void setColor(String color){
        this.color = color;
    }
    public void setWeight(int weight){
        this.weight = weight;
    }
    public void setThrownAttackValue(int thrownAttackValue){
        this.thrownAttackValue = thrownAttackValue;
    }
    public boolean equals(Item other)
    {
        if(other.getName().equals(name) && other.getColor().equals(color) && (other.getWeight() == weight) && (other.getThrownAttackValue() == thrownAttackValue))
        {
            return true;
        }
        else
        {
            return false;
        }
    }
    
    public String toString(){
        return name;
    }
}

import java.util.ArrayList;

public class Player {
    
    private ArrayList<Item> inventory;
    private String playerName;
    private int healthScore;
    
    public Player(String playerName, int healthScore){
        this.playerName = playerName;
        this.healthScore = healthScore;
        inventory = new ArrayList<Item>();
    }
    
    public ArrayList<Item> getInventory() {
        return inventory;
    }

    //allows player to pick up an item and adds it to the inventory
    public void pickUpItem(Item item) {
        inventory.add(item);
    }
    
    public String getName() {
        return playerName;
    }
    
    //allows player to drop an item, if that particular item is in the inventory
    //should return null if the item is not in the inventory in the first place
    public Item dropItem(Item item) {
        for (int i = 0; i < inventory.size(); i++) {
            if (inventory.get(i).equals(item)){
                inventory.remove(i);
                return item;
            }
        }
        return null;
    }
    
    
    //counts how many items are in the inventory and returns the count
    public int getItemCount()
    {
        return inventory.size();
    }
    
    //calculates the total weight of all the items in the inventory and returns it
    public double getTotalInventoryWeight() {
        double finalWeight = 0;
        for(int i = 0; i < inventory.size();i++)
        {
            finalWeight = finalWeight + ((inventory.get(i)).getWeight());
        }
        return finalWeight;
    }
    
    //returns true if the inventory contains a similar item (for example a sword, no matter color or size), false if it does not
    public boolean hasItem(Item item) {
            for (int i = 0; i < inventory.size(); i++) {
                if (inventory.get(i).getName().equals(item.getName())){
                    return true;
                }
            }
            return false;
    }
    
    //returns true if the inventory contains an exact item, false if it does not
    public boolean hasExactItem(Item item) {
         for(int i = 0; i < inventory.size();i++)
         {
             if(item.equals(inventory.get(i)))
             {
                 return true;
             }
         }
         return false;
    }
    
    //prints the name of the player and print the inventory of the player with each item printed on separate lines.
    public void printInventory() {
        System.out.println(playerName + "'s Inventory:");
        for (int i = 0; i < inventory.size();i++) {
            System.out.println("-" + inventory.get(i));
        }
    }
    //causes damage to player with certain amount of damage, subtracting from health
    public void takeDamage(int damage) {
        healthScore = healthScore - damage;
    }
    
    //heals player with certain amount of health, adding to health
    public void heal(int healthIncrease) {
        healthScore = healthScore + healthIncrease;
    }

}

import java.util.ArrayList;

public class Room {
    private String[] exits;
    private String roomName;
    private String roomDescription;
    private ArrayList<Item> roomItems;
    private ArrayList<Enemy> roomEnemies;
    
    public Room(String northExit, String eastExit, String southExit, String westExit, String roomName, String roomDescription){
    	exits = new String[]{northExit, eastExit, southExit, westExit};
    	this.roomName = roomName;
    	this.roomDescription = roomDescription;
    	roomItems = new ArrayList<Item>();
    	roomEnemies = new ArrayList<Enemy>();
    }
    
    public void addRoomItem(Item i) {
        roomItems.add(i);
    }
    
    public void addRoomEnemy(Enemy e) {
        roomEnemies.add(e);
    }
    
    public String getDescription() {
        return roomDescription;
    }
    
    public void setDescription(String roomDescription) {
        this.roomDescription = roomDescription;
    }
    
    public Item removeRoomItem(String name) {
        for (int i = 0; i < roomItems.size(); i++){
            if (roomItems.get(i).getName().equals(name)) {
                Item temp = roomItems.get(i);
                roomItems.remove(i);
                return temp;
            }
        }
        return null;
    }
    
    
    public boolean containsItem(String name){
        for (int i = 0; i < roomItems.size(); i++){
            if (roomItems.get(i).getName().equals(name)) {
                return true;
            }
        }
        return false;
    }
    public ArrayList<Item> getItems() {
        return roomItems;
    }
    
    public ArrayList<Enemy> getEnemies() {
        return roomEnemies;
    }
    
    public boolean hasEnemies() {
        return (roomEnemies.size() > 0);
    }
    
    public String toString() {
        String rs = "";
        rs = rs + roomName + "\n" + roomDescription + "\n" + "Exits: \n" + "To the North: " + exits[0] + "\n To the East: "  + exits[1] + "\n To the West: "  + exits[3] + "\n To the South: "   + exits[2];
        rs = rs + "\n In this room, you see: \n";
        for (int i = 0; i < roomItems.size(); i++) {
            rs = rs + roomItems.get(i) + "\n";
        }
        for (int i = 0; i < roomEnemies.size(); i++) {
            rs = rs + roomEnemies.get(i) + "\n";
        }
        return rs;
    }
    
    public String[] getExits()
    {
    	return exits;
    }
    
    public String getRoomName()
    {
    	return roomName;
    }
    
    public ArrayList<Item> getRoomItemsList()
{
	return roomItems;
}

public void removeRoomEnemy(String name) {
        roomEnemies.remove(0);
    }

}

    	else if (instruction.contains("east"))
    	{
    		possibleExit = exitNames[1];
    	}
    	else if (instruction.contains("south"))
    	{
    		possibleExit = exitNames[2];
    	}
    	else if (instruction.contains("west"))
    	{
    		possibleExit = exitNames[3];
    	}
    	else {
    	    System.out.println("***");
    	    System.out.println("Please move in a cardinal direction");
    	    System.out.println("***");
    	    return getCurrentRoom();
    	}
    	Item ankh = new Item("ankh","gold",3,3);
    	if (currentRoom == 2 && !player.hasItem(ankh) && instruction.contains("north")){
    	    System.out.println("***");
    	    System.out.println("You need an anchient key to cross the north portal");
    	    System.out.println("***");
    		return getCurrentRoom();
    	}
    	
    	if (currentRoom == 5 && getCurrentRoom().hasEnemies()  && instruction.contains("north")){
    	    System.out.println("***");
    	    System.out.println("You must kill the robot to go north");
    	    System.out.println("***");
    		return getCurrentRoom();
    	}
    	
    	if (possibleExit.equals("None"))
    	{
    	    System.out.println("***");
    	    System.out.println("There is no path this way. Try another direction.");
    	    System.out.println("***");
    		return getCurrentRoom();
    
    	}
    	else{
    		//find room you want to go to in room list
    		for (int i = 0; i < rooms.size(); i++)
    		{
    			if (rooms.get(i).getRoomName().equals(possibleExit))
    			{
    				//set currentRoom to that room
    				currentRoom = i;
    				return getCurrentRoom();
    			}
    		}
    		return getCurrentRoom();
	    }
		
    }

    
    public void pickUpItemFromRoom(String instruction) {
        Item thing = null;
        boolean command = instruction.contains("pick up");
        String item = instruction.substring(8).strip().toLowerCase();
        
        boolean itemInRoom = getCurrentRoom().containsItem(item);
        for (int i = 0; i < getCurrentRoom().getItems().size(); i++) {
            if (getCurrentRoom().getItems().get(i).getName().equals(item)) {
                thing = getCurrentRoom().getItems().get(i);
                break;
            }
        }
        if (thing == null) {
            System.out.println("***");
            System.out.println("You can't pick that up");
            System.out.println("***");
            return;
        }
        if (itemInRoom && command) {
            player.pickUpItem(thing);
            getCurrentRoom().removeRoomItem(item);
            System.out.println("***");
            System.out.println(item + " has been picked up");
            System.out.println("***");
        }
        
    }
    
    public void dropItemInRoom(String instruction) {
        Item thing = null;
        boolean command = instruction.contains("drop");
        String item = instruction.substring(5).toLowerCase();;
        for (int i = 0; i < player.getInventory().size(); i++) {
            if (player.getInventory().get(i).getName().equals(item)) {
                thing = player.getInventory().get(i);
                break;
            }
        }
        if (thing == null) {
            System.out.println("***");
            System.out.println("You don't have that item.");
            System.out.println("***");
            return;
        }
        boolean itemInInventory = player.hasItem(thing);
        if (itemInInventory && command) {
            player.dropItem(thing);
            getCurrentRoom().addRoomItem(thing);
            System.out.println("***");
            System.out.println(item + " has been dropped");
            System.out.println("***");
        }
    }
